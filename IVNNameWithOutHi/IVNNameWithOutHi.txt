	VARIABLE	: COUNTER	: NUMBER:1
	
	RETURN: STRING

	12	: SET: VARTEMP: ##TOKENVAR
	20	: IF: ##PRCN = ##COUNTER
	25	: BREAK
	30	: END IF
 	35	: SET		: COUNTER	: ##COUNTER + 1
	40	: END FOR
